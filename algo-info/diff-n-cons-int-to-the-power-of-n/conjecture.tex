\begin{algo}
	\In{$n \in \NNs$}
	\Out{?}
	
	\addalgoblank
	
	\Actions{
		Choisir $(n+1)$ naturels non nuls consécutifs : $k_1 < k_2 < \dots < k_{n+1}$.
		\\
		$L \Store [ k_1^n , k_2^n , \dots , k_{n+1}^n ]$
		\\
		\addalgoblank
		\While{$taille(L) \neq 1$}{
			$newL \Store \EmptyList$
			\\
			\ForRange*{$i$}{$1$}{$taille(L) - 1$}{
				Ajouter $(L[i + 1] - L[i])$ à droite de $newL$.
			}
			$L \Store newL$
		}
		\addalgoblank
		\Return{$L[1]$}
	}
\end{algo}
